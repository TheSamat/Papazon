В CLR есть возможность обойти доступность внутренних типов сборки другим, через механизм, называемый друзtественными сборками
(frendly assembly). Для этого используется атрибут InternalsVisibleTo из System.Runtime.ComplierServices:

using System;
using System.Runtime.ComplierServices;

[assembly:InternalsVisibleTo("Wintellect, PublicKey=12345678...90adcdef")]
[assembly:InternalsVisibleTo("Microsoft, PublicKey=b77a5c56...1934e089")]

internal sealed class SomeIntelnalType {...}
sealed class AnotherIntelnalType {...}
